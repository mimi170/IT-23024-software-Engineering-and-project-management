package controller;

import java.io.*;
import jakarta.servlet.*;
import jakarta.servlet.http.*;

public class StudentServlet extends HttpServlet {

    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String name = request.getParameter("name");
        String email = request.getParameter("email");
        String action = request.getParameter("action");

        response.setContentType("text/html");
        PrintWriter out = response.getWriter();

        out.println("<h2>Action: " + action + "</h2>");

        if(action.equals("Insert")){
            out.println("Inserted: " + name + " - " + email);
        }
        else if(action.equals("View")){
            out.println("<table border='1'>");
            out.println("<tr><th>Name</th><th>Email</th></tr>");
            out.println("<tr><td>Abdullah</td><td>ab@gmail.com</td></tr>");
            out.println("</table>");
        }
        else if(action.equals("Update")){
            out.println("Updated: " + name);
        }
        else if(action.equals("Delete")){
            out.println("Deleted: " + name);
        }
    }
}
